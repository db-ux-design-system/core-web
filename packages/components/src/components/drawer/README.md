## React

```tsx
import { useState } from "react";
import { DBButton, DBDrawer } from "../../../../../output/react/src";

const MyComponent = () => {
	const [open, setOpen] = useState<boolean>(false);
	return (
		<div>
			<DBDrawer
				open={open}
				onClose={() => {
					setOpen(false);
				}}
			>
				My Drawer content
			</DBDrawer>
			<DBButton
				onClick={() => {
					setOpen(true);
				}}
			>
				Open Me
			</DBButton>
		</div>
	);
};
```
