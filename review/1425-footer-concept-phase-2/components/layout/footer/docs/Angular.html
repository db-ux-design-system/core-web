Angularangular anchor link

For general installation and configuration take a look at the ngx-core-components package.

Use componentuse-component anchor link

Simplesimple anchor link

// app.component.ts
import { Component } from "@angular/core";
import { DBFooter } from "@db-ux/ngx-core-components";

@Component({
	selector: "app-root",
	imports: [DBFooter],
	template: `
		<db-footer>
			<div main>Footer Navigation</div>
			<div meta>Legal Links</div>
		</db-footer>
	`
})
export class AppComponent {}

With custom contentwith-custom-content anchor link

// app.component.ts
import { Component } from "@angular/core";
import { DBFooter, DBLink } from "@db-ux/ngx-core-components";

@Component({
	selector: "app-root",
	imports: [DBFooter, DBLink],
	template: `
		<db-footer>
			<div main>
				<ul>
					<li>
						<db-link href="#">About Us</db-link>
					</li>
					<li>
						<db-link href="#">Contact</db-link>
					</li>
					<li>
						<db-link href="#">Careers</db-link>
					</li>
				</ul>
			</div>
			<div meta>
				<ul>
					<li>
						<db-link href="#">Privacy Policy</db-link>
					</li>
					<li>
						<db-link href="#">Terms of Service</db-link>
					</li>
					<li>
						<db-link href="#">Imprint</db-link>
					</li>
				</ul>
			</div>
		</db-footer>
	`
})
export class AppComponent {}
