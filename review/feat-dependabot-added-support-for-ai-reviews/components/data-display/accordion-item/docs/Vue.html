Vuevue anchor link

For general installation and configuration look at the v-core-components package.

Use componentuse-component anchor link

With Slotswith-slots anchor link

<!-- App.vue -->
<script>
import { DBAccordion, DBAccordionItem } from "@db-ux/v-core-components";
</script>

<template>
	<DBAccordion>
		<DBAccordionItem>
			<template v-slot:headline>Title</template>
			Content
		</DBAccordionItem>
	</DBAccordion>
</template>

With attributeswith-attributes anchor link

<!-- App.vue -->
<script>
import { DBAccordion, DBAccordionItem } from "@db-ux/v-core-components";
</script>

<template>
	<DBAccordion>
		<DBAccordionItem headline="Title" content="Content"></DBAccordionItem>
	</DBAccordion>
</template>
