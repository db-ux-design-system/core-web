// App.tsx
import { useState } from "react";
import {
	DBButton,
	DBDrawer,
	DBDrawerHeader
} from "@db-ux/react-core-components";

const App = () => {
	const [open, setOpen] = useState<boolean>(false);
	return (
		<div>
			<DBButton
				onClick={() => {
					setOpen(true);
				}}
			>
				Open Me
			</DBButton>
			<DBDrawer
				open={open}
				onClose={() => {
					setOpen(false);
				}}
				header={
					<DBDrawerHeader closeButtonText="Close">
						Drawer Title
					</DBDrawerHeader>
				}
			>
				My Drawer content
			</DBDrawer>
		</div>
	);
};

export default App;
