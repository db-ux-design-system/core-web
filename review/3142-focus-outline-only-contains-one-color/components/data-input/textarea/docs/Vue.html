Generalgeneral anchor link

You cannot use a default value for textarea in vue. Use v-model:value or :value instead with a ref("My default value").

Use componentuse-component anchor link

<!-- App.vue -->
<script setup lang="ts">
import { ref } from "vue";
import { DBTextarea } from "@db-ui/v-components";

const textarea = ref("default value");
</script>

<template>
	<DBTextarea
		label="Textarea value"
		placeholder="Placeholder"
		message="Message"
		icon="user"
		name="textarevalue-name"
		v-model:value="textarea"
	/>
</template>
