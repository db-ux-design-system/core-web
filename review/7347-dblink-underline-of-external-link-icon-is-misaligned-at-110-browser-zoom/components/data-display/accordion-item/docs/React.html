// App.tsx
import { DBAccordion, DBAccordionItem } from "@db-ux/react-core-components";

const App = () => (
	<DBAccordion>
		<DBAccordionItem headline="Title">Content</DBAccordionItem>
	</DBAccordion>
);

// App.tsx
