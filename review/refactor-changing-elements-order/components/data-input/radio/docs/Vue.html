Vuevue anchor link

For general installation and configuration take a look at the v-components package.

Use componentuse-component anchor link

<!-- App.vue -->
<script setup lang="ts">
import { DBRadio } from "@db-ui/v-components";
import { ref } from "vue";
const radio = ref("");

const radioNames = ["X", "Y", "Z"];
</script>

<template>
	<ul>
		<li v-for="radioName in radioNames">
			<DBRadio
				@change="radio = radioName"
				name="radio-group"
				:value="radioName"
			>
				Radio {{ radioName }}
			</DBRadio>
		</li>
	</ul>
</template>
