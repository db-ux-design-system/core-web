How to use with Signal Formshow-to-use-with-signal-forms anchor link

Angular Signal Forms (Angular ≥ 21) are supported via Duck-Typing. The DBSelect component exposes a value ModelSignal that Angular's [formField] directive recognizes automatically.

//app.component.ts
import { DBSelect } from '@db-ux/ngx-core-components';
import { FormField } from '@angular/forms/signals';

@Component({
	// ...
	imports: [
		// ...,
		DBSelect,
		FormField
	],
	// ...
})

// form.component.ts
import { Component, signal } from "@angular/core";
import { form } from "@angular/forms/signals";

export class FormComponent {
	model = signal({ country: "" });
	myForm = form(this.model);

	onFormSubmit(): void {
		alert(JSON.stringify(this.model()));
	}
}

<!-- form.component.html -->
<form (submit)="onFormSubmit()">
	<db-select label="Country" [formField]="myForm.country">
		<option value="de">Germany</option>
		<option value="at">Austria</option>
		<option value="ch">Switzerland</option>
	</db-select>
	<button type="submit">Submit</button>
</form>

<h2>Output</h2>
<dl>
	<dt>select's value</dt>
	<dd>{{ myForm.country().value() || "No selection" }}</dd>
</dl>

How to use with Reactive Formshow-to-use-with-reactive-forms anchor link

Third party controls require a ControlValueAccessor to function with angular forms. Our select component implements this interface so you can use it like any other native element with reactive Forms:

//app.component.ts
import { ReactiveFormsModule } from '@angular/forms';

@Component({
	// ...
	imports: [
		// ...,
		ReactiveFormsModule
	],
	// ...
})

<!-- form.component.html -->
<form [formGroup]="form" (submit)="onFormSubmit()">
	<db-select label="Select" formControlName="select">
		<option value="test1">Test1</option>
		<option value="test2">Test2</option>
		<option value="test3">Test3</option>
	</db-select>
</form>

<h2>Output</h2>
<dl>
	<dt>select's value</dt>
	<dd>
		{{ form.get("select")?.value ? form.get("select")?.value : "No
		selection" }}
	</dd>
</dl>

// form.component.ts
export class FormComponent {
	form = new FormGroup({
		select: new FormControl("test1")
	});

	onFormSubmit(): void {
		alert(JSON.stringify(this.form.value));
	}
}

How to use with Template Driven Formshow-to-use-with-template-driven-forms anchor link

Third party controls require a ControlValueAccessor to function with angular forms. Our select component implements this interface so you can use it like any other native element with ngModel:

//app.component.ts
import { FormsModule } from '@angular/forms';

@Component({
	// ...
	imports: [
		// ...,
		FormsModule
	],
	// ...
})

<!-- form.component.html -->
<form>
	<DBSelect ngDefaultControl [(ngModel)]="select" label="Select">
		<option value="test1">Test1</option>
		<option value="test2">Test2</option>
		<option value="test3">Test3</option>
	</DBSelect>
	<DBButton type="button" variant="brand" (click)="showValues()"
		>Get select value</DBButton
	>
</form>

<h2>Output</h2>
<dl>
	<dt>select's value</dt>
	<dd>{{ select ? select : "No selection" }}</dd>
</dl>

// form.component.ts
export class FormComponent {
	select = "";
	showValues(): void {
		alert(JSON.stringify({ select: this.select }));
	}
}
