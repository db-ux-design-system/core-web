Vuevue anchor link

For general installation and configuration take a look at the v-core-components package.

Use componentuse-component anchor link

<!-- App.vue -->
<script setup lang="ts">
import { DBRadio } from "@db-ux/v-core-components";
import { _ref } from "vue";
const radio = _ref("");

const radioNames = ["X", "Y", "Z"];
</script>

<template>
	<fieldset>
		<legend>Radio group example</legend>
		<DBRadio
			v-for="radioName in radioNames"
			:key="radioName"
			@change="radio = radioName"
			name="radio-group"
			:value="radioName"
			:checked="radio === radioName"
		>
			Radio {{ radioName }}
		</DBRadio>
	</fieldset>
</template>
