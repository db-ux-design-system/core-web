How to use with Signal Formshow-to-use-with-signal-forms anchor link

Angular Signal Forms (Angular ≥ 21) are partially supported for radio buttons. Due to a known limitation, radio button groups don't integrate well with [formField] yet. Use signal-based (change) bindings instead:

//app.component.ts
import { DBRadio } from '@db-ux/ngx-core-components';

@Component({
	// ...
	imports: [
		// ...,
		DBRadio
	],
	// ...
})

// form.component.ts
import { Component, signal } from "@angular/core";

export class FormComponent {
	selectedRadio = signal("");

	handleRadioChange(event: Event): void {
		this.selectedRadio.set((event.target as HTMLInputElement).value);
	}

	onFormSubmit(): void {
		alert(JSON.stringify({ radio: this.selectedRadio() }));
	}
}

<!-- form.component.html -->
<form (submit)="onFormSubmit()">
	<db-radio
		(change)="handleRadioChange($event)"
		value="option-1"
		name="radio-group"
		label="Option 1"
	></db-radio>
	<db-radio
		(change)="handleRadioChange($event)"
		value="option-2"
		name="radio-group"
		label="Option 2"
	></db-radio>
	<db-radio
		(change)="handleRadioChange($event)"
		value="option-3"
		name="radio-group"
		label="Option 3"
	></db-radio>
	<button type="submit">Submit</button>
</form>

<h2>Output</h2>
<dl>
	<dt>radio's value</dt>
	<dd>{{ selectedRadio() || "No radio set" }}</dd>
</dl>

How to use with Template Driven Formshow-to-use-with-template-driven-forms anchor link

Third party controls require a ControlValueAccessor to function with angular forms. Adding an ngDefaultControl attribute will allow them to use that directive. Further information

//app.component.ts
import { FormsModule } from '@angular/forms';

@Component({
	// ...
	imports: [
		// ...,
		FormsModule
	],
	// ...
})

<!-- form.component.html -->
<form>
	<DBRadio ngDefaultControl [(ngModel)]="radio">Label</DBRadio>
	<DBButton type="button" variant="brand" (click)="showValues()"
		>Get radio value</DBButton
	>
</form>

<h2>Output</h2>
<dl>
	<dt>radio's value</dt>
	<dd>{{ radio ? radio : "No radio set" }}</dd>
</dl>

// form.component.ts
export class FormComponent {
	radio = "";
	showValues(): void {
		alert(JSON.stringify({ radio: this.radio }));
	}
}
