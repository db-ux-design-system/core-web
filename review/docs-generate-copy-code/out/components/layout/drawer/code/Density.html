How to use this in angular
<db-drawer
  class="db-density-functional"
  [withCloseButton]="true"
  [open]="open"
  (onClose)="toggleDrawer(false)"
>
  Functional
</db-drawer>
How to use this in html
<dialog class="db-drawer">
  <article class="db-drawer-container db-density-functional">
    <header class="db-drawer-header">
      <div class="db-drawer-header-text"></div>
      <button
        class="db-button button-close-drawer"
        data-icon="cross"
        data-variant="ghost"
        data-no-text="true"
      >
        Close Button
      </button>
    </header>
    <div class="db-drawer-content">Functional</div>
  </article>
</dialog>
