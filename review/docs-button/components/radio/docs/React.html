App.tsx

// App.tsx
import { useState } from "react";
import { DBRadio } from "@db-ui/react-components";
const App = () => (
<ul>
{["X", "Y", "Z"].map((radioName) => (
<li key={radioName}>
<DBRadio
name="radio-group"
onChange={() => {
setRadio(radioName);
}}
>
Radio {radioName}
</DBRadio>
</li>
))}
</ul>
);
export default App;
